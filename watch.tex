\addcontentsline{toc}{subsection}{watch}
\headerline{\large watch\index{watch}}{\emph{repeatedly run a command}}{}
\hrule
\vspace{4mm}

\paragraph{Description}
\indentpar \raggedright \textrm{Runs the given command every two seconds.}\\

\paragraph{Usage}
\indentpar watch \textit{command}

\paragraph{Arguments}
\indentpar \argumentline{command}{the command to be run repeatedly}

\paragraph{Examples}

\indentpar watch ls\\
\indentpar watch wc file.txt
