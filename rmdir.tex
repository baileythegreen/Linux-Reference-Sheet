\addcontentsline{toc}{subsection}{rmdir}
\headerline{\large rmdir\index{rmdir}}{\emph{remove directories if empty}}{}
\hrule
\vspace{4mm}

\paragraph{Description}
\indentpar \raggedright \textrm{Removes a directory if and only if it is empty.}\\

\paragraph{Usage}
\indentpar rmdir \textit{directory [directory2...]}

\paragraph{Arguments}
\indentpar \argumentline{directory}{\texttt{directory} to be removed}\\
\indentpar \argumentline{directory2...}{additional directories (optional)}

\paragraph{Examples}
\indentpar rmdir directory
